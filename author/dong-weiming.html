    抽象工厂模式

    python的例子

    from abc import ABCMeta

class StandardFactory(object):
    '''这就是那个抽象工厂'''
    @staticmethod
    def get_factory(factory):
        '''根据参数找到对实际操作的工厂'''
        if factory == 'cat':
            return CatFactory()
        elif factory == 'dog':
            return DogFactory()
        raise TypeError('Unknown Factory.')

    python的例子

    这里实现一个打印操作系统名字的功能

    def someFunctionality(self):
        raise NotImplemented()

    # 根据初始化参数传入实现的产品类

    def someFunctionality(self):

    dispatch_table = {
    'get': get,

    func()
else:
